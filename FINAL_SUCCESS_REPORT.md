## 💡 Usage Guide

### Toast Notifications
```tsx
import toast from 'react-hot-toast';

// Success
toast.success('Action completed!');

// Error
toast.error('Something went wrong');

// Loading
const id = toast.loading('Processing...');
toast.success('Done!', { id });

// Custom
toast('Custom message', {
  icon: '🚀',
  duration: 4000,
});
```

import SkeletonLoader, {

<SkeletonStats />
